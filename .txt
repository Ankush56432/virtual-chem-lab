// ========================================
// Toast Notification Manager
// ========================================
class ToastManager {
  constructor() {
    this.container = document.getElementById('toastContainer');
  }
  
  show(message, type = 'success', duration = 4000) {
    const toast = document.createElement('div');
    toast.className = `toast ${type}`;
    toast.setAttribute('role', 'alert');
    toast.setAttribute('aria-live', 'assertive');
    
    const icons = {
      success: '‚úì',
      error: '‚úï',
      warning: '‚ö†',
      info: '‚Ñπ'
    };
    
    toast.innerHTML = `
      <span class="toast-icon" aria-hidden="true">${icons[type] || icons.info}</span>
      <span class="toast-message">${this.escapeHtml(message)}</span>
      <button class="toast-close" aria-label="Close notification">√ó</button>
    `;
    
    this.container.appendChild(toast);
    
    const closeBtn = toast.querySelector('.toast-close');
    closeBtn.addEventListener('click', () => this.remove(toast));
    
    if (duration > 0) {
      setTimeout(() => this.remove(toast), duration);
    }
    
    return toast;
  }
  
  escapeHtml(text) {
    const div = document.createElement('div');
    div.textContent = text;
    return div.innerHTML;
  }
  
  remove(toast) {
    toast.style.opacity = '0';
    toast.style.transform = 'translateX(400px)';
    setTimeout(() => {
      if (toast.parentNode) {
        toast.remove();
      }
    }, 300);
  }
  
  success(message) { return this.show(message, 'success'); }
  error(message) { return this.show(message, 'error', 6000); }
  warning(message) { return this.show(message, 'warning', 5000); }
  info(message) { return this.show(message, 'info'); }
}

// ========================================
// Loading Manager
// ========================================
class LoadingManager {
  constructor() {
    this.overlay = document.getElementById('loadingOverlay');
    this.count = 0;
  }
  
  show() {
    this.count++;
    if (this.overlay) {
      this.overlay.classList.add('active');
      this.overlay.setAttribute('aria-busy', 'true');
    }
  }
  
  hide() {
    this.count = Math.max(0, this.count - 1);
    if (this.count === 0 && this.overlay) {
      this.overlay.classList.remove('active');
      this.overlay.setAttribute('aria-busy', 'false');
    }
  }
}

// ========================================
// Auth Service (Connected to Backend)
// ========================================
class AuthService {
  constructor() {
    this.apiBase = `${API_BASE_URL}/auth`;
  }
  
  async register(data) {
    try {
      const response = await fetch(`${this.apiBase}/register`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        credentials: 'include',
        body: JSON.stringify(data)
      });
      
      const result = await response.json();
      
      if (!response.ok) {
        throw new Error(result.error || 'Registration failed');
      }
      
      // Store user in localStorage for quick access
      if (result.user) {
        localStorage.setItem('user', JSON.stringify(result.user));
      }
      
      return result;
    } catch (error) {
      console.error('Registration error:', error);
      throw error;
    }
  }
  
  async login(email, password) {
    try {
      const response = await fetch(`${this.apiBase}/login`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        credentials: 'include',
        body: JSON.stringify({ email, password })
      });
      
      const result = await response.json();
      
      if (!response.ok) {
        throw new Error(result.error || 'Login failed');
      }
      
      // Store user in localStorage
      if (result.user) {
        localStorage.setItem('user', JSON.stringify(result.user));
      }
      
      return result;
    } catch (error) {
      console.error('Login error:', error);
      throw error;
    }
  }
  
  async logout() {
    try {
      await fetch(`${this.apiBase}/logout`, {
        method: 'POST',
        credentials: 'include'
      });
      
      localStorage.removeItem('user');
    } catch (error) {
      console.error('Logout error:', error);
      // Clear local data even if request fails
      localStorage.removeItem('user');
    }
  }
  
  async getCurrentUser() {
    try {
      const response = await fetch(`${this.apiBase}/me`, {
        credentials: 'include'
      });
      
      if (!response.ok) {
        throw new Error('Not authenticated');
      }
      
      const result = await response.json();
      
      if (result.user) {
        localStorage.setItem('user', JSON.stringify(result.user));
        return result.user;
      }
      
      return null;
    } catch (error) {
      localStorage.removeItem('user');
      return null;
    }
  }
  
  getLocalUser() {
    const user = localStorage.getItem('user');
    return user ? JSON.parse(user) : null;
  }
}

const toast = new ToastManager();

document.addEventListener('DOMContentLoaded', async () => {
  // Set year in footer

  let isLogin = true;
  let isLoggedIn = false;

  // DOM Elements
  const authOverlay = document.getElementById('authOverlay');
  const authClose = document.getElementById('authClose');
  const loginBtn = document.getElementById('loginBtn');
  const userAccountWrapper = document.getElementById('userAccountWrapper');
  const userAvatarBtn = document.getElementById('userAvatarBtn');
  const accountDropdown = document.getElementById('accountDropdown');
  const dropdownLogout = document.getElementById('dropdownLogout');
  const switchModeBtn = document.getElementById('switchMode');
  const authForm = document.getElementById('authForm');
  const togglePassword = document.getElementById('togglePassword');

  const confirmPasswordGroup = document.getElementById('confirmPasswordGroup');
  const loginOptions = document.getElementById('loginOptions');
  const termsText = document.getElementById('termsText');

  const nameInput = document.getElementById('nameInput');
  const emailInput = document.getElementById('emailInput');
  const passwordInput = document.getElementById('passwordInput');
  const confirmPasswordInput = document.getElementById('confirmPasswordInput');

  const brandHeading = document.getElementById('brandHeading');
  const brandDescription = document.getElementById('brandDescription');
  const formTitle = document.getElementById('formTitle');
  const formSubtitle = document.getElementById('formSubtitle');
  const submitText = document.getElementById('submitText');
  const switchText = document.getElementById('switchText');
  const submitBtn = document.getElementById('submitBtn');

  // ========================================
  // Auth UI Functions
  // ========================================
  function loadUserDropdownData() {
    const user = authService.getLocalUser();
    if (!user) return;

    const initial = user.name ? user.name.charAt(0).toUpperCase() : 'U';
    
    const userAvatarInitial = document.getElementById('userAvatarInitial');
    const dropdownAvatarInitial = document.getElementById('dropdownAvatarInitial');
    const dropdownUserName = document.getElementById('dropdownUserName');
    const dropdownUserEmail = document.getElementById('dropdownUserEmail');

    if (userAvatarInitial) userAvatarInitial.textContent = initial;
    if (dropdownAvatarInitial) dropdownAvatarInitial.textContent = initial;
    if (dropdownUserName) dropdownUserName.textContent = user.name || 'User';
    if (dropdownUserEmail) dropdownUserEmail.textContent = user.email || 'user@email.com';
  }

  function setLoggedInUI() {
    isLoggedIn = true;
    if (loginBtn) loginBtn.classList.add('hidden');
    if (userAccountWrapper) userAccountWrapper.classList.remove('hidden');
    if (authOverlay) authOverlay.classList.remove('active');
    loadUserDropdownData();
  }

  function setLoggedOutUI(showModal = false) {
    isLoggedIn = false;
    if (loginBtn) loginBtn.classList.remove('hidden');
    if (userAccountWrapper) userAccountWrapper.classList.add('hidden');
    if (accountDropdown) accountDropdown.classList.remove('show');
    if (authOverlay && showModal) authOverlay.classList.add('active');
  }

  async function checkLoginStatus() {
    try {
      loading.show();
      const user = await authService.getCurrentUser();
      if (user) {
        setLoggedInUI();
        return true;
      }
      setLoggedOutUI(false);
      return false;
    } catch (error) {
      console.error('Login check error:', error);
      setLoggedOutUI(false);
      return false;
    } finally {
      loading.hide();
    }
  }

  // Check initial login status
  await checkLoginStatus();

  // ========================================
  // Form Validation
  // ========================================
  function clearErrors() {
    document.querySelectorAll('.error-message').forEach(el => {
      el.classList.remove('show');
      el.textContent = '';
    });
    document.querySelectorAll('.form-input').forEach(el => {
      el.classList.remove('error');
      el.removeAttribute('aria-invalid');
    });
  }

  function clearInputs() {
    if (nameInput) nameInput.value = '';
    if (emailInput) emailInput.value = '';
    if (passwordInput) passwordInput.value = '';
    if (confirmPasswordInput) confirmPasswordInput.value = '';
  }

  function showError(input, errorElement, message) {
    if (!input || !errorElement) return;
    input.classList.add('error');
    input.setAttribute('aria-invalid', 'true');
    errorElement.textContent = message;
    errorElement.classList.add('show');
  }

  function validateForm() {
    clearErrors();
    let isValid = true;
    let firstErrorInput = null;

    if (!isLogin && nameInput && !nameInput.value.trim()) {
      showError(nameInput, document.getElementById('nameError'), 'Name is required');
      if (!firstErrorInput) firstErrorInput = nameInput;
      isValid = false;
    }

    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (emailInput && !emailInput.value.trim()) {
      showError(emailInput, document.getElementById('emailError'), 'Email is required');
      if (!firstErrorInput) firstErrorInput = emailInput;
      isValid = false;
    } else if (emailInput && !emailRegex.test(emailInput.value)) {
      showError(emailInput, document.getElementById('emailError'), 'Please enter a valid email');
      if (!firstErrorInput) firstErrorInput = emailInput;
      isValid = false;
    }

    if (passwordInput && !passwordInput.value) {
      showError(passwordInput, document.getElementById('passwordError'), 'Password is required');
      if (!firstErrorInput) firstErrorInput = passwordInput;
      isValid = false;
    } else if (passwordInput && passwordInput.value.length < 8) {
      showError(passwordInput, document.getElementById('passwordError'), 'Password must be at least 8 characters');
      if (!firstErrorInput) firstErrorInput = passwordInput;
      isValid = false;
    }

    if (!isLogin && confirmPasswordInput) {
      if (!confirmPasswordInput.value) {
        showError(confirmPasswordInput, document.getElementById('confirmPasswordError'), 'Please confirm your password');
        if (!firstErrorInput) firstErrorInput = confirmPasswordInput;
        isValid = false;
      } else if (passwordInput && passwordInput.value !== confirmPasswordInput.value) {
        showError(confirmPasswordInput, document.getElementById('confirmPasswordError'), 'Passwords do not match');
        if (!firstErrorInput) firstErrorInput = confirmPasswordInput;
        isValid = false;
      }
    }

    // Focus on first error
    if (firstErrorInput) {
      firstErrorInput.focus();
    }

    return isValid;
  }

  function updateAuthUI() {
    if (isLogin) {
      if (nameGroup) nameGroup.classList.add('hidden');
      if (confirmPasswordGroup) confirmPasswordGroup.classList.add('hidden');
      if (loginOptions) loginOptions.classList.remove('hidden');
      if (termsText) termsText.classList.remove('show');

      if (brandHeading) brandHeading.textContent = 'Welcome Back!';
      if (brandDescription) brandDescription.textContent = 'Access state-of-the-art computational chemistry tools and continue your learning journey.';
      if (formTitle) formTitle.textContent = 'Sign In';
      if (formSubtitle) formSubtitle.textContent = 'Enter your credentials to access your account';
      if (submitText) submitText.textContent = 'Sign In';
      if (switchText) switchText.textContent = "Don't have an account? ";
      if (switchModeBtn) switchModeBtn.textContent = 'Sign Up';
    } else {
      if (nameGroup) nameGroup.classList.remove('hidden');
      if (confirmPasswordGroup) confirmPasswordGroup.classList.remove('hidden');
      if (loginOptions) loginOptions.classList.add('hidden');
      if (termsText) termsText.classList.add('show');

      if (brandHeading) brandHeading.textContent = 'Start Your Journey';
      if (brandDescription) brandDescription.textContent = 'Join thousands exploring computational chemistry in our virtual laboratory.';
      if (formTitle) formTitle.textContent = 'Create Account';
      if (formSubtitle) formSubtitle.textContent = 'Fill in your information to get started';
      if (submitText) submitText.textContent = 'Create Account';
      if (switchText) switchText.textContent = 'Already have an account? ';
      if (switchModeBtn) switchModeBtn.textContent = 'Sign In';
    }
  }

  // ========================================
  // Auth Form Submission
  // ========================================
  async function handleAuthSubmit(e) {
    e.preventDefault();
    if (!validateForm()) return;

    const email = emailInput ? emailInput.value.trim() : '';
    const password = passwordInput ? passwordInput.value : '';
    const name = !isLogin && nameInput ? nameInput.value.trim() : '';

    if (!email || !password) return;

    if (submitBtn) {
      submitBtn.disabled = true;
      submitText.textContent = 'Processing...';
    }

    loading.show();

    try {
      if (isLogin) {
        await authService.login(email, password);
        toast.success('Welcome back! Login successful.');
      } else {
        await authService.register({ email, password, name });
        toast.success('Account created successfully! Welcome to VirtualChem Labs.');
      }
      
      setLoggedInUI();
      clearInputs();
    } catch (error) {
      console.error('Auth error:', error);
      toast.error(error.message || 'An error occurred. Please try again.');
    } finally {
      loading.hide();
      if (submitBtn) {
        submitBtn.disabled = false;
        submitText.textContent = isLogin ? 'Sign In' : 'Create Account';
      }
    }
  }

  // ========================================
  // Event Listeners
  // ========================================
  if (loginBtn) {
    loginBtn.addEventListener('click', () => {
      authOverlay.classList.add('active');
      setTimeout(() => {
        if (emailInput) emailInput.focus();
      }, 100);
    });
  }

  if (authClose) {
    authClose.addEventListener('click', () => {
      authOverlay.classList.remove('active');
    });
  }

  if (authOverlay) {
    authOverlay.addEventListener('click', (e) => {
      if (e.target === authOverlay) {
        authOverlay.classList.remove('active');
      }
    });
  }

  document.addEventListener('keydown', (e) => {
    if (e.key === 'Escape' && authOverlay && authOverlay.classList.contains('active')) {
      authOverlay.classList.remove('active');
    }
  });

  if (togglePassword && passwordInput) {
    togglePassword.addEventListener('click', () => {
      const newType = passwordInput.type === 'password' ? 'text' : 'password';
      passwordInput.type = newType;
      if (confirmPasswordInput) confirmPasswordInput.type = newType;
      togglePassword.textContent = newType === 'password' ? 'üëÅÔ∏è' : 'üôà';
      togglePassword.setAttribute('aria-label', `${newType === 'password' ? 'Show' : 'Hide'} password`);

  if (switchModeBtn) {
    switchModeBtn.addEventListener('click', () => {

      clearErrors();

        if (isLogin && emailInput) {

          nameInput.focus();

    const indicators = sliderWrapper.querySelectorAll('.indicator');

    const nextBtn = sliderWrapper.querySelector('.slider-next');
    const progressBar = sliderWrapper.querySelector('.slider-progress-bar');
    
    let currentSlide = 0;
    let isAutoPlaying = true;
    let autoPlayInterval;
    let progressInterval;
    const slideInterval = 7000;
    const progressStep = 100 / (slideInterval / 50);

    function showSlide(index) {
      slides.forEach(slide => slide.classList.remove('active'));
      indicators.forEach(indicator => indicator.classList.remove('active'));

      currentSlide = (index + slides.length) % slides.length;
      slides[currentSlide].classList.add('active');
      indicators[currentSlide].classList.add('active');

      if (progressBar) progressBar.style.width = '0%';

      clearInterval(progressInterval);
      let progress = 0;
      progressInterval = setInterval(() => {
        progress += progressStep;
        if (progressBar) {
          progressBar.style.width = Math.min(progress, 100) + '%';
        }
        if (progress >= 100) {
          clearInterval(progressInterval);
        }
      }, 50);
    }

    function nextSlide() {
      showSlide(currentSlide + 1);
    }

    function prevSlide() {
      showSlide(currentSlide - 1);
    }

    function startAutoPlay() {
      if (isAutoPlaying) {
        autoPlayInterval = setInterval(nextSlide, slideInterval);
      }
    }

    function stopAutoPlay() {
      clearInterval(autoPlayInterval);
      clearInterval(progressInterval);
    }

    function resetAutoPlay() {
      stopAutoPlay();
      if (isAutoPlaying) {
        startAutoPlay();
      }
    }

    if (prevBtn) {
      prevBtn.addEventListener('click', () => {
        prevSlide();
        resetAutoPlay();
      });
    }

    if (nextBtn) {
      nextBtn.addEventListener('click', () => {
        nextSlide();
        resetAutoPlay();
      });
    }

    indicators.forEach((indicator, index) => {
      indicator.addEventListener('click', () => {
        showSlide(index);
        resetAutoPlay();
      });
    });

    document.addEventListener('keydown', (e) => {
      if (!sliderWrapper) return;
      if (e.key === 'ArrowLeft') {
        prevSlide();
        resetAutoPlay();
      } else if (e.key === 'ArrowRight') {
        nextSlide();
        resetAutoPlay();
      }
    });

    sliderWrapper.addEventListener('mouseenter', () => {
      isAutoPlaying = false;
      stopAutoPlay();
    });

    sliderWrapper.addEventListener('mouseleave', () => {
      isAutoPlaying = true;
      startAutoPlay();
    });

    document.addEventListener('visibilitychange', () => {
      if (document.hidden) {
        stopAutoPlay();
      } else if (isAutoPlaying) {
        startAutoPlay();
      }
    });

    let touchStartX = 0;
    let touchEndX = 0;

    sliderWrapper.addEventListener('touchstart', (e) => {
      touchStartX = e.changedTouches[0].screenX;
    }, { passive: true });

    sliderWrapper.addEventListener('touchend', (e) => {
      touchEndX = e.changedTouches[0].screenX;
      handleSwipe();
    }, { passive: true });

    function handleSwipe() {
      const swipeThreshold = 50;
      if (touchEndX < touchStartX - swipeThreshold) {
        nextSlide();
        resetAutoPlay();
      }
      if (touchEndX > touchStartX + swipeThreshold) {
        prevSlide();
        resetAutoPlay();
      }
    }

    showSlide(0);
    startAutoPlay();
  }
